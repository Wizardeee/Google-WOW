def compute_degree_centrality(G):
    # Calculate degree centrality for target ranking
    centrality = nx.degree_centrality(G)
    top_accounts = sorted(
        centrality.items(), 
        key=lambda x: x[1], 
        reverse=True
    )[:10]
    return top_accounts
